N, M, K = map(int, input().split())
A = list(map(int, input().split()))
B = list(map(int, input().split()))
prefA = [0] * (N + 1)
for i in range(N):
    prefA[i + 1] = prefA[i] + A[i]
prefB = [0] * (M + 1)
for i in range(M):
    prefB[i + 1] = prefB[i] + B[i]
ans = 0
b = M
for a in range(N + 1):
    if prefA[a] > K:
        break
    remaining = K - prefA[a]
    while b > 0 and prefB[b] > remaining:
        b -= 1
    if a + b > ans:
        ans = a + b
print(ans)
